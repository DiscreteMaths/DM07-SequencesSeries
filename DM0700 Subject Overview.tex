\section{Sigma Notation}

{Sigma Notation}

\textbf{Three Important Summation Identities}\\
\textit{index term = i , Number of terms = n }

\begin{itemize}
\item Identity 1
\[ \sum^{i=n}_{i=1} 1 = n \]
\item Identity 2  
\[ \sum^{i=n}_{i=1} i = \frac{n(n+1)}{2} \]
\item Identity 3 
\[ \sum^{i=n}_{i=1} i^2 = \frac{(2n+1)(n+1)(n)}{6} \]
\end{itemize}
